```sql
CREATE TABLE `prj_project` (

    `code` VARCHAR(50) NOT NULL COMMENT '项目编码',

CREATE TABLE `prj_project_member` (
    `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '主键ID',

    `var_value` TEXT COMMENT '变量值(加密存储)',
    `is_secret` TINYINT NOT NULL DEFAULT 0 COMMENT '是否敏感变量',
    `description` VARCHAR(255) COMMENT '描述',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

    UNIQUE KEY `uk_project_env_key` (`project_id`, `env`, `var_key`)
```

```java
// com.drp.project.entity.Project
@Entity
@Table(name = "prj_project")
@Data
@EqualsAndHashCode(callSuper = true)
public class Project extends BaseEntity {

    @Column(nullable = false, length = 100)
    private String name;

    @Column(nullable = false, unique = true, length = 50)
    private String code;

    @Column(length = 500)
    private String description;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false, length = 20)
    private ProjectType type;

    @Column(name = "git_url", length = 500)
    private String gitUrl;

    @Enumerated(EnumType.STRING)
    @Column(name = "git_auth_type", length = 20)
    private GitAuthType gitAuthType;

    @Column(name = "git_credentials", columnDefinition = "TEXT")
    private String gitCredentials;  // 加密存储

    @Column(name = "default_branch", length = 100)
    private String defaultBranch = "master";

    @Column(name = "release_branches", length = 500)
    private String releaseBranches;

    @Column(nullable = false)
    private Integer status = 1;

    // 辅助方法
    public List<String> getReleaseBranchList() {
        if (StringUtils.isBlank(releaseBranches)) {
            return Collections.emptyList();
        }
        return Arrays.asList(releaseBranches.split(","));
    }
}
```

```java
// com.drp.project.service.ProjectServiceImpl
@Service
@RequiredArgsConstructor
public class ProjectServiceImpl implements ProjectService {

    private final ProjectRepository projectRepository;
    private final AesEncryptUtil encryptUtil;

    @Override
    @Transactional
    public ProjectDTO create(ProjectCreateRequest request) {
        // 检查编码唯一性
        if (projectRepository.existsByCode(request.getCode())) {
            throw new BusinessException("项目编码已存在");
        }

        Project project = new Project();
        BeanUtils.copyProperties(request, project);

        // 加密Git凭证
        if (StringUtils.isNotBlank(project.getGitCredentials())) {
            project.setGitCredentials(encryptUtil.encrypt(project.getGitCredentials()));
        }

        project.setCreateBy(getCurrentUsername());
        project = projectRepository.save(project);

        return toDTO(project);
    }

    @Override
    public ProjectDTO getById(Long id) {
        Project project = projectRepository.findById(id)
            .orElseThrow(() -> new BusinessException(ResultCode.PROJECT_NOT_FOUND));
        return toDTO(project);
    }

    private ProjectDTO toDTO(Project project) {
        ProjectDTO dto = new ProjectDTO();
        BeanUtils.copyProperties(project, dto);
        // 解密Git凭证（脱敏）
        if (StringUtils.isNotBlank(project.getGitCredentials())) {
            dto.setHasCredentials(true);
        }
        return dto;
    }
}
```

### 5.3 环境变量服务

```java
// com.drp.project.service.EnvVariableService
@Service
@RequiredArgsConstructor
public class EnvVariableService {

    private final EnvVariableRepository repository;
    private final AesEncryptUtil encryptUtil;

    @Transactional
    public EnvVariableDTO create(Long projectId, EnvVariableCreateRequest request) {
        // 验证项目存在
        validateProjectExists(projectId);

        // 检查变量是否已存在
        if (repository.existsByProjectIdAndEnvAndVarKey(projectId, request.getEnv(), request.getVarKey())) {
            throw new BusinessException("该变量已存在");
        }

        EnvVariable variable = new EnvVariable();
        variable.setProjectId(projectId);
        variable.setEnv(request.getEnv());
        variable.setVarKey(request.getVarKey());
        variable.setDescription(request.getDescription());
        variable.setSecret(request.getSecret());

        // 加密敏感变量
        if (request.getSecret() && StringUtils.isNotBlank(request.getVarValue())) {
            variable.setVarValue(encryptUtil.encrypt(request.getVarValue()));
        } else {
            variable.setVarValue(request.getVarValue());
        }

        variable = repository.save(variable);
        return toDTO(variable);
    }

    /**
     * 获取解密后的变量值（供构建/部署时调用）
     */
    public String getDecryptedValue(Long variableId) {
        EnvVariable variable = repository.findById(variableId)
            .orElseThrow(() -> new BusinessException("变量不存在"));

        if (variable.getSecret() && StringUtils.isNotBlank(variable.getVarValue())) {
            return encryptUtil.decrypt(variable.getVarValue());
        }
        return variable.getVarValue();
    }
}
```

### 5.4 控制器

```java
// com.drp.project.controller.ProjectController
@RestController
@RequestMapping("/api/projects")
@RequiredArgsConstructor
public class ProjectController {

    private final ProjectService projectService;

    @GetMapping
    public Result<PageResponse<ProjectDTO>> list(PageRequest request) {
        return Result.success(projectService.list(request));
    }

    @GetMapping("/{id}")
    public Result<ProjectDTO> getById(@PathVariable Long id) {
        return Result.success(projectService.getById(id));
    }

    @PostMapping
    public Result<ProjectDTO> create(@RequestBody @Valid ProjectCreateRequest request) {
        return Result.success(projectService.create(request));
    }

    @PutMapping("/{id}")
    public Result<ProjectDTO> update(@PathVariable Long id,
                                      @RequestBody @Valid ProjectUpdateRequest request) {
        return Result.success(projectService.update(id, request));
    }

    @DeleteMapping("/{id}")
    public Result<Void> delete(@PathVariable Long id) {
        projectService.delete(id);
        return Result.success("删除成功");
    }
}
```

## 6. 前端API

```typescript
// src/api/project.ts
import request from './request'

export interface Project {
  id: number
  name: string
  code: string
  description: string
  type: 'JAVA_MAVEN' | 'JAVA_GRADLE' | 'NODE' | 'MIXED'
  gitUrl: string
  gitAuthType: 'HTTPS' | 'SSH'
  hasCredentials: boolean
  defaultBranch: string
  releaseBranches: string
  status: number
}

export interface EnvVariable {
  id: number
  projectId: number
  env: 'dev' | 'test' | 'prod'
  varKey: string
  varValue: string
  secret: boolean
  description: string
}

export const projectApi = {
  list(params: any) {
    return request.get<any>('/projects', { params })
  },

  getById(id: number) {
    return request.get<Project>(`/projects/${id}`)
  },

  create(data: Partial<Project>) {
    return request.post<Project>('/projects', data)
  },

  update(id: number, data: Partial<Project>) {
    return request.put<Project>(`/projects/${id}`, data)
  },

  delete(id: number) {
    return request.delete(`/projects/${id}`)
  },

  // 环境变量
  getVariables(projectId: number, env?: string) {
    return request.get<EnvVariable[]>(`/projects/${projectId}/variables`, {
      params: { env }
    })
  },

  createVariable(projectId: number, data: Partial<EnvVariable>) {
    return request.post<EnvVariable>(`/projects/${projectId}/variables`, data)
  },

  updateVariable(projectId: number, id: number, data: Partial<EnvVariable>) {
    return request.put<EnvVariable>(`/projects/${projectId}/variables/${id}`, data)
  },

  deleteVariable(projectId: number, id: number) {
    return request.delete(`/projects/${projectId}/variables/${id}`)
  }
}
```

```vue
<template>

      </el-table-column>
    </el-table>

    <!-- 创建/编辑弹窗 -->
    <el-dialog v-model="dialogVisible" :title="dialogTitle" width="600px">
      <el-form ref="formRef" :model="form" :rules="rules" label-width="100px">
        <el-form-item label="项目名称" prop="name">
          <el-input v-model="form.name" />
        </el-form-item>
        <el-form-item label="项目编码" prop="code">

        </el-form-item>
        <el-form-item label="项目类型" prop="type">
          <el-select v-model="form.type">
            <el-option label="Java Maven" value="JAVA_MAVEN" />
            <el-option label="Java Gradle" value="JAVA_GRADLE" />
            <el-option label="Node.js" value="NODE" />
            <el-option label="混合项目" value="MIXED" />
          </el-select>
        </el-form-item>
        <el-form-item label="Git仓库">
          <el-input v-model="form.gitUrl" placeholder="https://github.com/xxx/xxx.git" />
        </el-form-item>
        <el-form-item label="默认分支">
          <el-input v-model="form.defaultBranch" placeholder="master" />
        </el-form-item>
      </el-form>
      <template #footer>
        <el-button @click="dialogVisible = false">取消</el-button>
        <el-button type="primary" @click="handleSubmit">确定</el-button>
      </template>
    </el-dialog>
  </div>

<script setup lang="ts">
import { ref, reactive } from 'vue'
import { projectApi, type Project } from '@/api/project'
import { ElMessage, ElMessageBox } from 'element-plus'

const loading = ref(false)
const projects = ref<Project[]>([])
const dialogVisible = ref(false)
const dialogTitle = ref('')
const formRef = ref()

const typeMap = {
  JAVA_MAVEN: 'Java Maven',
  JAVA_GRADLE: 'Java Gradle',
  NODE: 'Node.js',
  MIXED: '混合项目'
}

const form = reactive<Partial<Project>>({
  id: undefined,
  name: '',
  code: '',
  type: 'JAVA_MAVEN',
  gitUrl: '',
  defaultBranch: 'master'
})

const rules = {
  name: [{ required: true, message: '请输入项目名称', trigger: 'blur' }],
  code: [
    { required: true, message: '请输入项目编码', trigger: 'blur' },
    { pattern: /^[a-z][a-z0-9-]*$/, message: '编码格式：小写字母、数字、中划线' }
  ],
  type: [{ required: true, message: '请选择项目类型', trigger: 'change' }]
}

async function loadProjects() {
  loading.value = true
  try {
    projects.value = await projectApi.list({})
  } finally {
    loading.value = false
  }
}

function handleCreate() {
  dialogTitle.value = '新建项目'
  Object.assign(form, { id: undefined, name: '', code: '', type: 'JAVA_MAVEN', gitUrl: '', defaultBranch: 'master' })
  dialogVisible.value = true
}

function handleEdit(row: Project) {
  dialogTitle.value = '编辑项目'
  Object.assign(form, row)
  dialogVisible.value = true
}

async function handleSubmit() {
  const valid = await formRef.value?.validate().catch(() => false)
  if (!valid) return

  if (form.id) {
    await projectApi.update(form.id, form)
    ElMessage.success('更新成功')
  } else {
    await projectApi.create(form)
    ElMessage.success('创建成功')
  }
  dialogVisible.value = false
  loadProjects()
}

async function handleDelete(row: Project) {
  await ElMessageBox.confirm(`确定删除项目「${row.name}」吗？`, '提示')
  await projectApi.delete(row.id)
  ElMessage.success('删除成功')
  loadProjects()
}

function handleVariable(row: Project) {
  // 跳转到变量管理页面
}

loadProjects()
</script>
```
