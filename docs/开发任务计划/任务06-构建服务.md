### 5.2 流水线执行器

```java
// com.drp.build.pipeline.PipelineExecutor
@Component
@RequiredArgsConstructor
@Slf4j
public class PipelineExecutor {

    private final BuildLogHandler logHandler;
    private final GitService gitService;
    private final MavenExecutor mavenExecutor;
    private final NpmExecutor npmExecutor;

    public BuildRecord execute(Long buildId, Pipeline pipeline, String workDir) {
        BuildRecord record = buildRecordRepository.findById(buildId).orElseThrow();
        PipelineContext context = new PipelineContext(buildId, workDir);

        try {
            record.setStatus(BuildStatus.RUNNING);
            record.setStartTime(LocalDateTime.now());
            buildRecordRepository.save(record);

            for (Stage stage : pipeline.getStages()) {
                if (!stage.isEnabled()) continue;

                logHandler.sendLog(buildId, String.format("\n[Stage] %s\n", stage.getName()));
                boolean stageResult = executeStage(stage, context, buildId);

                if (!stageResult && stage.isFailFast()) {
                    throw new BuildException("Stage [" + stage.getName() + "] failed");
                }
            }

            // 构建成功
            record.setStatus(BuildStatus.SUCCESS);
            record.setEndTime(LocalDateTime.now());
            record.setDuration(
                Duration.between(record.getStartTime(), record.getEndTime()).toMillis()
            );

        } catch (Exception e) {
            log.error("构建失败", e);
            record.setStatus(BuildStatus.FAIL);
            record.setErrorMessage(e.getMessage());
            record.setEndTime(LocalDateTime.now());
            logHandler.sendLog(buildId, String.format("\n[ERROR] %s\n", e.getMessage()));
        }

        return buildRecordRepository.save(record);
    }

    private boolean executeStage(Stage stage, PipelineContext context, Long buildId) {
        for (Step step : stage.getSteps()) {
            try {
                logHandler.sendLog(buildId, String.format("[Step] %s\n", step.getName()));
                String output = executeStep(step, context);
                logHandler.sendLog(buildId, output + "\n");
            } catch (Exception e) {
                logHandler.sendLog(buildId, String.format("[ERROR] %s\n", e.getMessage()));
                return false;
            }
        }
        return true;
    }

    private String executeStep(Step step, PipelineContext context) {
        return switch (step.getType()) {
            case GIT_CLONE -> gitService.clone(context);
            case MAVEN -> mavenExecutor.execute(step.getCommand(), context.getWorkDir());
            case NPM -> npmExecutor.execute(step.getCommand(), context.getWorkDir());
            default -> throw new BuildException("Unknown step type: " + step.getType());
        };
    }
}
```

### 5.3 Git 服务

```java
// com.drp.build.service.GitService
@Component
@RequiredArgsConstructor
public class GitService {

    private final AesEncryptUtil encryptUtil;

    public void clone(Project project, String branch, String workDir) throws GitAPIException {
        String credentials = encryptUtil.decrypt(project.getGitCredentials());

        CredentialsProvider provider = new UsernamePasswordCredentialsProvider(
            extractUsername(credentials),
            extractPassword(credentials)
        );

        Git.cloneRepository()
            .setURI(project.getGitUrl())
            .setBranch(branch)
            .setDirectory(new File(workDir))
            .setCredentialsProvider(provider)
            .setCloneAllBranches(false)
            .call();
    }

    public String getCommitId(String workDir) throws GitAPIException {
        try (Git git = Git.open(new File(workDir))) {
            return git.log().setMaxCount(1).call().iterator().next().getName();
        }
    }

    public String getCommitMessage(String workDir) throws GitAPIException {
        try (Git git = Git.open(new File(workDir))) {
            return git.log().setMaxCount(1).call().iterator().next().getFullMessage();
        }
    }
}
```

### 6.1 构建日志处理器

```java
// com.drp.build.websocket.BuildLogHandler
@Component
@RequiredArgsConstructor
public class BuildLogHandler {

    private final BuildLogSessionManager sessionManager;

    /**
     * 发送日志到WebSocket
     */
    public void sendLog(Long buildId, String content) {
        sessionManager.sendToSession(buildId, content);
    }

    /**
     * 发送JSON格式日志
     */
    public void sendLogJson(Long buildId, String type, Object data) {
        try {
            String json = new ObjectMapper().writeValueAsString(
                Map.of("type", type, "data", data, "timestamp", System.currentTimeMillis())
            );
            sessionManager.sendToSession(buildId, json);
        } catch (JsonProcessingException e) {
            log.error("序列化日志失败", e);
        }
    }

    /**
     * 标记阶段开始
     */
    public void sendStageStart(Long buildId, String stageName) {
        sendLogJson(buildId, "stage_start", Map.of("name", stageName));
    }

    /**
     * 标记阶段结束
     */
    public void sendStageEnd(Long buildId, String stageName, boolean success) {
        sendLogJson(buildId, "stage_end", Map.of("name", stageName, "success", success));
    }
}
```

## 7. 版本号生成

```java
// com.drp.build.service.VersionService
@Service
@RequiredArgsConstructor
public class VersionService {

    private final BuildRecordRepository buildRecordRepository;

    /**
     * 生成版本号: v{日期}-{commitId}-{buildNumber}
     * 例: v20260416-a1b2c3d-001
     */
    public String generateVersion(String commitId, Long projectId) {
        String date = LocalDate.now().format(DateTimeFormatter.ofPattern("yyyyMMdd"));

        // 获取当日构建次数
        LocalDateTime startOfDay = LocalDate.now().atStartOfDay();
        long todayBuildCount = buildRecordRepository.countByProjectIdAndCreateTimeAfter(projectId, startOfDay);
        int buildNumber = (int) (todayBuildCount + 1);

        String commitPrefix = commitId != null && commitId.length() >= 7
            ? commitId.substring(0, 7)
            : "0000000";

        return String.format("v%s-%s-%03d", date, commitPrefix, buildNumber);
    }
}
```

```vue
<script setup lang="ts">
import { ref, reactive, onMounted, onUnmounted } from 'vue'
import { buildApi, projectApi } from '@/api'
import { ElMessage } from 'element-plus'
import LogViewer from '@/components/log/LogViewer.vue'
import SockJS from 'sockjs-client'
import Stomp from 'stompjs'

const loading = ref(false)
const builds = ref<any[]>([])
const projects = ref<any[]>([])
const branches = ref<string[]>([])
const buildLogs = ref<string[]>([])
const logDialogVisible = ref(false)
const triggerDialogVisible = ref(false)
const triggerFormRef = ref()

let stompClient: Stomp.Client | null = null

const query = reactive({ projectId: undefined, status: '' })

const triggerForm = reactive({
  projectId: undefined,
  branch: ''
})

const triggerRules = {
  projectId: [{ required: true, message: '请选择项目', trigger: 'change' }],
  branch: [{ required: true, message: '请选择分支', trigger: 'change' }]
}

const statusType: Record<string, string> = {
  PENDING: 'info',
  RUNNING: 'warning',
  SUCCESS: 'success',
  FAIL: 'danger',
  CANCEL: 'info'
}

const statusText: Record<string, string> = {
  PENDING: '待执行',
  RUNNING: '执行中',
  SUCCESS: '成功',
  FAIL: '失败',
  CANCEL: '已取消'
}

async function loadBuilds() {
  loading.value = true
  try {
    builds.value = await buildApi.list(query)
  } finally {
    loading.value = false
  }
}

async function loadProjects() {
  projects.value = await projectApi.list({})
}

async function onProjectChange(projectId: number) {
  const project = projects.value.find(p => p.id === projectId)
  if (project) {
    branches.value = project.releaseBranches?.split(',') || [project.defaultBranch]
  }
}

async function handleTriggerSubmit() {
  const valid = await triggerFormRef.value?.validate().catch(() => false)
  if (!valid) return

  await buildApi.trigger(triggerForm)
  ElMessage.success('构建已触发')
  triggerDialogVisible.value = false
  loadBuilds()
}

function handleViewLog(row: any) {
  logDialogVisible.value = true
  buildLogs.value = []
  connectWebSocket(row.id)
}

function connectWebSocket(buildId: number) {
  const socket = new SockJS(`/ws/build/${buildId}/log`)
  stompClient = Stomp.over(socket)

  stompClient.connect({}, () => {
    stompClient?.subscribe(`/topic/build/${buildId}`, (message) => {
      buildLogs.value.push(message.body)
    })
  })
}

function disconnectWebSocket() {
  if (stompClient) {
    stompClient.disconnect()
    stompClient = null
  }
}

async function handleCancel(row: any) {
  await buildApi.cancel(row.id)
  ElMessage.success('构建已取消')
  loadBuilds()
}

onMounted(() => {
  loadProjects()
  loadBuilds()
})

onUnmounted(() => {
  disconnectWebSocket()
})
</script>
```
