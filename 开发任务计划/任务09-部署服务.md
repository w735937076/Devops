## 1. 任务目标

实现部署管理模块，支持一键发布、多种部署策略（单机/滚动/蓝绿）、一键回滚、健康检查、部署日志实时查看等核心功能。

### 5.1 单机部署策略

```java
// com.drp.deploy.strategy.SingleDeployStrategy
@Component
@RequiredArgsConstructor
public class SingleDeployStrategy implements DeployStrategy {

    private final SshService sshService;
    private final DeployLogHandler logHandler;
    private final HealthCheckService healthCheckService;
    private final BackupExecutor backupExecutor;

    @Override
    public DeployType getType() {
        return DeployType.SINGLE;
    }

    @Override
    public DeployResult deploy(DeployContext context) {
        DeployRecord record = context.getRecord();
        Server server = context.getServer();
        Version version = context.getVersion();

        try {
            updateProgress(record, 0, "开始部署");

            // 1. 备份当前版本
            updateProgress(record, 10, "备份当前版本");
            logHandler.sendLog(record.getId(), "[1/6] 备份当前版本...\n");
            backupExecutor.backup(server, context);

            // 2. 上传新版本
            updateProgress(record, 30, "上传构建产物");
            logHandler.sendLog(record.getId(), "[2/6] 上传构建产物...\n");
            uploadArtifact(server, version, context);

            // 3. 停止服务
            updateProgress(record, 50, "停止服务");
            logHandler.sendLog(record.getId(), "[3/6] 停止服务...\n");
            stopService(server, context);

            // 4. 启动新版本
            updateProgress(record, 70, "启动服务");
            logHandler.sendLog(record.getId(), "[4/6] 启动服务...\n");
            startService(server, version, context);

            // 5. 健康检查
            updateProgress(record, 85, "健康检查");
            logHandler.sendLog(record.getId(), "[5/6] 健康检查...\n");
            boolean healthy = healthCheckService.check(server, context);

            if (!healthy) {
                throw new DeployException("健康检查失败");
            }

            // 6. 完成
            updateProgress(record, 100, "部署完成");
            logHandler.sendLog(record.getId(), "[6/6] 部署完成!\n");

            return DeployResult.success(record.getId());

        } catch (Exception e) {
            log.error("部署失败", e);
            logHandler.sendLog(record.getId(), "[ERROR] " + e.getMessage() + "\n");
            return DeployResult.fail(e.getMessage());
        }
    }

    @Override
    public RollbackResult rollback(RollbackContext context) {
        DeployRecord record = context.getRecord();
        Server server = context.getServer();

        try {
            logHandler.sendLog(record.getId(), "开始回滚...\n");

            // 恢复备份
            backupExecutor.restore(server, context);

            // 重启服务
            restartService(server, context);

            // 健康检查
            if (!healthCheckService.check(server, context)) {
                throw new DeployException("回滚后健康检查失败");
            }

            logHandler.sendLog(record.getId(), "回滚完成!\n");
            return RollbackResult.success();

        } catch (Exception e) {
            return RollbackResult.fail(e.getMessage());
        }
    }

    private void stopService(Server server, DeployContext context) {
        String command = String.format(
            "cd %s && ./scripts/stop.sh %s",
            server.getWorkDir(),
            context.getProject().getName()
        );
        sshService.execute(server, command);
    }

    private void startService(Server server, Version version, DeployContext context) {
        String jarName = context.getProject().getName() + ".jar";
        String command = String.format(
            "cd %s && nohup ./scripts/start.sh %s %s > /dev/null 2>&1 &",
            server.getWorkDir(),
            jarName,
            version.getVersion()
        );
        sshService.execute(server, command);
    }
}
```

## 6. 部署前检查

```java
// com.drp.deploy.service.PreCheckService
@Service
@RequiredArgsConstructor
public class PreCheckService {

    private final SshService sshService;
    private final ServerRepository serverRepository;
    private final ProjectServerRepository projectServerRepository;

    public PreCheckResult check(Long projectId, Long versionId, String env) {
        PreCheckResult result = new PreCheckResult();
        List<PreCheckItem> items = new ArrayList<>();

        // 1. 检查服务器连通性
        items.add(checkServerConnection(projectId, env));

        // 2. 检查磁盘空间
        items.add(checkDiskSpace(projectId, env));

        // 3. 检查端口占用
        items.add(checkPortAvailable(projectId, env));

        // 4. 检查上一个版本健康状态
        items.add(checkLastVersionHealth(projectId, env));

        result.setItems(items);
        result.setAllPassed(items.stream().allMatch(PreCheckItem::isPassed));

        return result;
    }

    private PreCheckItem checkServerConnection(Long projectId, String env) {
        PreCheckItem item = new PreCheckItem();
        item.setName("服务器连通性");

        List<ProjectServer> servers = projectServerRepository.findByProjectIdAndEnv(projectId, env);
        try {
            for (ProjectServer ps : servers) {
                Server server = ps.getServer();
                String output = sshService.execute(server, "echo 'ok'");
                if (!output.contains("ok")) {
                    throw new Exception("服务器响应异常");
                }
            }
            item.setPassed(true);
            item.setMessage("所有服务器连接正常");
        } catch (Exception e) {
            item.setPassed(false);
            item.setMessage("服务器连接失败: " + e.getMessage());
        }

        return item;
    }

    private PreCheckItem checkDiskSpace(Long projectId, String env) {
        PreCheckItem item = new PreCheckItem();
        item.setName("磁盘空间检查");

        List<ProjectServer> servers = projectServerRepository.findByProjectIdAndEnv(projectId, env);
        try {
            for (ProjectServer ps : servers) {
                Server server = ps.getServer();
                String output = sshService.execute(server, "df -h / | tail -1 | awk '{print $5}'");
                int usedPercent = Integer.parseInt(output.replace("%", "").trim());

                if (usedPercent > 90) {
                    item.setPassed(false);
                    item.setMessage(String.format("服务器 %s 磁盘使用率 %d%% > 90%%",
                        server.getHostname(), usedPercent));
                    return item;
                }
            }
            item.setPassed(true);
            item.setMessage("磁盘空间充足");
        } catch (Exception e) {
            item.setPassed(false);
            item.setMessage("磁盘检查失败: " + e.getMessage());
        }

        return item;
    }
}
```

## 7. 健康检查服务

```java
// com.drp.deploy.service.HealthCheckService
@Service
@RequiredArgsConstructor
public class HealthCheckService {

    private final SshService sshService;

    /**
     * 健康检查
     * 1. 检查进程是否存在
     * 2. 检查端口是否监听
     * 3. 检查健康接口 (可选)
     */
    public boolean check(Server server, DeployContext context) {
        Project project = context.getProject();
        String appName = project.getName();

        // 等待服务启动 (默认30s)
        int maxWaitSeconds = 30;
        int waited = 0;

        while (waited < maxWaitSeconds) {
            try {
                Thread.sleep(2000);
                waited += 2;

                // 检查进程
                if (!checkProcess(server, appName)) {
                    continue;
                }

                // 检查端口
                if (!checkPort(server, project.getPort())) {
                    continue;
                }

                // 检查健康接口 (如果有)
                if (project.getHealthCheckUrl() != null) {
                    if (!checkHealthEndpoint(server, project.getHealthCheckUrl())) {
                        continue;
                    }
                }

                return true;
            } catch (Exception e) {
                log.debug("健康检查等待中...", e);
            }
        }

        return false;
    }

    private boolean checkProcess(Server server, String appName) {
        try {
            String command = String.format("pgrep -f '%s.*jar'", appName);
            String output = sshService.execute(server, command);
            return output != null && !output.isEmpty();
        } catch (Exception e) {
            return false;
        }
    }

    private boolean checkPort(Server server, int port) {
        try {
            String command = String.format("netstat -tln | grep ':%d '", port);
            String output = sshService.execute(server, command);
            return output != null && output.contains(String.valueOf(port));
        } catch (Exception e) {
            return false;
        }
    }
}
```

```java
        return Result.success(deployService.healthCheck(id));
```

```vue
import Stomp from 'stompjs'

const formRef = ref()
const projects = ref<any[]>([])
const versions = ref<any[]>([])
const deploying = ref(false)
const currentDeploy = ref<any>(null)
const deployLogs = ref<string[]>([])
const previewVisible = ref(false)
const previewData = ref<any>({})

let stompClient: Stomp.Client | null = null

const form = reactive({
  projectId: undefined,
  versionId: undefined,
  env: 'dev',
  strategy: 'SINGLE'
})

const rules = {
  projectId: [{ required: true, message: '请选择项目', trigger: 'change' }],
  versionId: [{ required: true, message: '请选择版本', trigger: 'change' }],
  env: [{ required: true, message: '请选择环境', trigger: 'change' }]
}

const statusType: Record<string, string> = {
  PENDING: 'info',
  RUNNING: 'warning',
  SUCCESS: 'success',
  FAIL: 'danger',
  ROLLBACK: 'warning'
}

const statusText: Record<string, string> = {
  PENDING: '待执行',
  RUNNING: '执行中',
  SUCCESS: '成功',
  FAIL: '失败',
  ROLLBACK: '回滚中'
}

const progressStatus = computed(() => {
  if (currentDeploy.value?.status === 'SUCCESS') return 'success'
  if (currentDeploy.value?.status === 'FAIL') return 'exception'
  return undefined
})

async function onProjectChange(projectId: number) {
  // 加载项目版本列表
  versions.value = await deployApi.getVersions(projectId)
}

async function handlePreview() {
  const valid = await formRef.value?.validate().catch(() => false)
  if (!valid) return

  previewData.value = await deployApi.preview(form)
  previewVisible.value = true
}

async function handleConfirmDeploy() {
  previewVisible.value = false

  await ElMessageBox.confirm(
    `确认部署到 ${form.env} 环境？`,
    '确认部署',
    { type: 'warning' }
  )

  await handleDeploy()
}

async function handleDeploy() {
  const valid = await formRef.value?.validate().catch(() => false)
  if (!valid) return

  if (form.env === 'prod') {
    await ElMessageBox.confirm(
      '即将部署到生产环境，请再次确认！',
      '生产环境部署',
      { confirmButtonText: '确认部署', cancelButtonText: '取消', type: 'warning' }
    )
  }

  deploying.value = true
  try {
    currentDeploy.value = await deployApi.trigger(form)
    connectWebSocket(currentDeploy.value.id)
    ElMessage.success('部署已触发')
  } finally {
    deploying.value = false
  }
}

function connectWebSocket(deployId: number) {
  const socket = new SockJS(`/ws/deploy/${deployId}/log`)
  stompClient = Stomp.over(socket)

  stompClient.connect({}, () => {
    stompClient?.subscribe(`/topic/deploy/${deployId}`, (message) => {
      deployLogs.value.push(message.body)

      // 解析进度更新
      try {
        const data = JSON.parse(message.body)
        if (data.type === 'progress' && currentDeploy.value) {
          currentDeploy.value.progress = data.data.progress
          currentDeploy.value.status = data.data.status
        }
      } catch {}
    })
  })
}

function disconnectWebSocket() {
  if (stompClient) {
    stompClient.disconnect()
    stompClient = null
  }
}

onMounted(async () => {
  projects.value = await projectApi.list({})
})

onUnmounted(() => {
  disconnectWebSocket()
})
</script>
```
