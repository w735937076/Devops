### 5.2 企业微信渠道

```java
// com.drp.notify.channel.WeComChannel
@Component
@RequiredArgsConstructor
public class WeComChannel implements NotificationChannel {

    private final RestTemplate restTemplate;

    @Override
    public NotificationChannelType getChannelType() {
        return NotificationChannelType.WECOM;
    }

    @Override
    public SendResult send(NotifyMessage message, JSONObject config) {
        String webhookUrl = config.getString("webhookUrl");

        // 构建消息内容
        Map<String, Object> body = new HashMap<>();
        body.put("msgtype", "markdown");

        Map<String, Object> markdown = new HashMap<>();
        markdown.put("content", buildMarkdownContent(message));
        body.put("markdown", markdown);

        try {
            ResponseEntity<String> response = restTemplate.postForEntity(
                webhookUrl, body, String.class);

            if (response.getStatusCode().is2xxSuccessful()) {
                return SendResult.success();
            } else {
                return SendResult.fail("发送失败: " + response.getBody());
            }
        } catch (Exception e) {
            return SendResult.fail("发送异常: " + e.getMessage());
        }
    }

    @Override
    public boolean test(JSONObject config) {
        try {
            NotifyMessage testMsg = NotifyMessage.builder()
                .title("测试通知")
                .content("这是一条测试消息")
                .build();
            SendResult result = send(testMsg, config);
            return result.isSuccess();
        } catch (Exception e) {
            return false;
        }
    }

    private String buildMarkdownContent(NotifyMessage message) {
        StringBuilder sb = new StringBuilder();
        sb.append("### ").append(message.getTitle()).append("\n\n");
        sb.append(message.getContent()).append("\n\n");
        if (message.getProjectName() != null) {
            sb.append("- **项目**: ").append(message.getProjectName()).append("\n");
        }
        if (message.getEnv() != null) {
            sb.append("- **环境**: ").append(message.getEnv()).append("\n");
        }
        if (message.getVersion() != null) {
            sb.append("- **版本**: ").append(message.getVersion()).append("\n");
        }
        sb.append("- **时间**: ").append(LocalDateTime.now().format(
            DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss"))).append("\n");
        return sb.toString();
    }
}
```

### 5.4 通知服务

```java
// com.drp.notify.service.NotificationService
@Service
@RequiredArgsConstructor
public class NotificationServiceImpl implements NotificationService {

    private final NotificationConfigRepository configRepository;
    private final NotificationRecordRepository recordRepository;
    private final Map<NotificationChannelType, NotificationChannel> channelMap;

    @Async
    @Override
    public void sendNotification(NotifyEvent event) {
        // 查找匹配的配置
        List<NotificationConfig> configs = findConfigs(event);

        for (NotificationConfig config : configs) {
            if (!config.getEnabled()) continue;

            NotificationChannel channel = channelMap.get(config.getChannel());
            if (channel == null) {
                log.warn("未找到通知渠道: {}", config.getChannel());
                continue;
            }

            // 构建消息
            NotifyMessage message = buildMessage(event, config);

            // 发送并记录
            JSONObject channelConfig = JSON.parseObject(config.getConfig());
            SendResult result = channel.send(message, channelConfig);

            // 保存发送记录
            saveRecord(config, message, result);
        }
    }

    private List<NotificationConfig> findConfigs(NotifyEvent event) {
        // 优先查找项目级配置，再查找全局配置
        List<NotificationConfig> configs = new ArrayList<>();

        if (event.getProjectId() != null) {
            configs.addAll(configRepository.findByProjectIdAndEventAndEnabled(
                event.getProjectId(), event.getEvent(), true));
        }

        configs.addAll(configRepository.findByProjectIdIsNullAndEventAndEnabled(
            event.getEvent(), true));

        return configs;
    }

    private NotifyMessage buildMessage(NotifyEvent event, NotificationConfig config) {
        return NotifyMessage.builder()
            .title(buildTitle(event))
            .content(buildContent(event))
            .projectName(event.getProjectName())
            .env(event.getEnv())
            .version(event.getVersion())
            .operator(event.getOperator())
            .build();
    }
}
```

## 6. 事件监听

```java
// com.drp.notify.service.NotifyEventListener
@Component
@RequiredArgsConstructor
public class NotifyEventListener {

    private final NotificationService notificationService;

    @EventListener
    public void onBuildEvent(BuildEvent event) {
        NotifyEvent notifyEvent = NotifyEvent.builder()
            .event(event.getStatus() == "SUCCESS" ? "BUILD_SUCCESS" : "BUILD_FAIL")
            .projectId(event.getProjectId())
            .projectName(event.getProjectName())
            .version(event.getVersion())
            .operator(event.getOperator())
            .build();

        notificationService.sendNotification(notifyEvent);
    }

    @EventListener
    public void onDeployEvent(DeployEvent event) {
        String notifyEvent = switch (event.getStatus()) {
            case "SUCCESS" -> "DEPLOY_SUCCESS";
            case "FAIL" -> "DEPLOY_FAIL";
            case "ROLLBACK_SUCCESS" -> "ROLLBACK_SUCCESS";
            case "ROLLBACK_FAIL" -> "ROLLBACK_FAIL";
            default -> null;
        };

        if (notifyEvent != null) {
            NotifyEvent notify = NotifyEvent.builder()
                .event(notifyEvent)
                .projectId(event.getProjectId())
                .projectName(event.getProjectName())
                .env(event.getEnv())
                .version(event.getVersion())
                .operator(event.getOperator())
                .build();

            notificationService.sendNotification(notify);
        }
    }
}
```
