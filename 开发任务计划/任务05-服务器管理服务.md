## 5. SSH 连接服务

### 5.1 SSH 服务核心代码

```java
// com.drp.server.service.SshService
@Component
@RequiredArgsConstructor
public class SshService {

    private final AesEncryptUtil encryptUtil;

    /**
     * 创建SSH连接
     */
    public SSHClient createClient(Server server) throws IOException {
        SSHClient client = new SSHClient();

        // 配置算法
        client.addHostKeyVerifier(new PromiscuousVerifier());

        // 连接
        client.connect(server.getHostname(), server.getPort());

        // 认证
        if (server.getPrivateKey() != null) {
            // 私钥认证
            String privateKey = encryptUtil.decrypt(server.getPrivateKey());
            String passphrase = server.getPrivateKeyPassphrase() != null
                ? encryptUtil.decrypt(server.getPrivateKeyPassphrase())
                : null;

            Resource resource = new ByteArrayResource(privateKey.getBytes());
            IdentityKey identityKey = IdentityKey.fromPrivateKey(resource, passphrase);
            client.authPublickey(server.getUsername(), identityKey);
        } else {
            // 密码认证
            String password = encryptUtil.decrypt(server.getPassword());
            client.authPassword(server.getUsername(), password);
        }

        return client;
    }

    /**
     * 执行命令
     */
    public String execute(Server server, String command) throws IOException {
        try (SSHClient client = createClient(server)) {
            Session session = client.startSession();
            try {
                Session.CommandResult result = session.exec(command);
                String output = IOUtils.readFully(result.getInputStream(), StandardCharsets.UTF_8);
                result.join(30, TimeUnit.SECONDS);
                return output;
            } finally {
                session.close();
            }
        }
    }

    /**
     * 测试连接
     */
    public ConnectionTestResult testConnection(Server server) {
        ConnectionTestResult result = new ConnectionTestResult();
        result.setHostname(server.getHostname());
        result.setPort(server.getPort());

        long start = System.currentTimeMillis();
        try {
            String output = execute(server, "echo 'connection_test'");
            if (output.contains("connection_test")) {
                result.setSuccess(true);
                result.setMessage("连接成功");
            }
        } catch (Exception e) {
            result.setSuccess(false);
            result.setMessage("连接失败: " + e.getMessage());
        }

        result.setCostMs(System.currentTimeMillis() - start);
        return result;
    }
}
```

## 6. 服务器服务

```java
// com.drp.server.service.ServerServiceImpl
@Service
@RequiredArgsConstructor
@Slf4j
public class ServerServiceImpl implements ServerService {

    private final ServerRepository serverRepository;
    private final SshService sshService;
    private final AesEncryptUtil encryptUtil;

    @Override
    @Transactional
    public ServerDTO create(ServerCreateRequest request) {
        Server server = new Server();
        BeanUtils.copyProperties(request, server);

        // 加密敏感信息
        if (StringUtils.isNotBlank(request.getPassword())) {
            server.setPassword(encryptUtil.encrypt(request.getPassword()));
        }
        if (StringUtils.isNotBlank(request.getPrivateKey())) {
            server.setPrivateKey(encryptUtil.encrypt(request.getPrivateKey()));
        }
        if (StringUtils.isNotBlank(request.getPrivateKeyPassphrase())) {
            server.setPrivateKeyPassphrase(encryptUtil.encrypt(request.getPrivateKeyPassphrase()));
        }

        server = serverRepository.save(server);
        return toDTO(server);
    }

    @Override
    public ConnectionTestResult testConnection(Long id) {
        Server server = serverRepository.findById(id)
            .orElseThrow(() -> new BusinessException(ResultCode.SERVER_NOT_FOUND));
        return sshService.testConnection(server);
    }

    @Override
    @Transactional
    public void updateHeartbeat(Long id) {
        serverRepository.findById(id).ifPresent(server -> {
            server.setLastHeartbeat(LocalDateTime.now());
            server.setStatus(ServerStatus.ONLINE);
            serverRepository.save(server);
        });
    }
}
```

## 7. 定时心跳检测

```java
// com.drp.server.task.HeartbeatTask
@Component
@RequiredArgsConstructor
public class ServerHeartbeatTask {

    private final ServerRepository serverRepository;
    private final SshService sshService;

    @Scheduled(fixedRate = 60000)  // 每分钟执行
    public void checkHeartbeat() {
        List<Server> servers = serverRepository.findByStatus(ServerStatus.ONLINE);

        for (Server server : servers) {
            try {
                sshService.execute(server, "echo 'heartbeat'");
                server.setLastHeartbeat(LocalDateTime.now());
            } catch (Exception e) {
                log.warn("服务器 {} 心跳检测失败: {}", server.getHostname(), e.getMessage());
                server.setStatus(ServerStatus.OFFLINE);
            }
            serverRepository.save(server);
        }
    }
}
```

### 8.2 API 封装

```typescript
// src/api/server.ts
import request from './request'

export interface Server {
  id: number
  name: string
  hostname: string
  port: number
  username: string
  authType: 'password' | 'key'
  workDir: string
  backupDir: string
  status: number
  lastHeartbeat: string
}

export interface ConnectionTestResult {
  hostname: string
  port: number
  success: boolean
  message: string
  costMs: number
}

export const serverApi = {
  list(params?: any) {
    return request.get<Server[]>('/servers', { params })
  },

  getById(id: number) {
    return request.get<Server>(`/servers/${id}`)
  },

  create(data: Partial<Server>) {
    return request.post<Server>('/servers', data)
  },

  update(id: number, data: Partial<Server>) {
    return request.put<Server>(`/servers/${id}`, data)
  },

  delete(id: number) {
    return request.delete(`/servers/${id}`)
  },

  test(id: number) {
    return request.post<ConnectionTestResult>(`/servers/${id}/test`)
  }
}
```
