// com.drp.test.IntegrationTest
@SpringBootTest
@ActiveProfiles("test")
public class DeployIntegrationTest {

    @Autowired
    private DeployService deployService;

    @Autowired
    private ProjectService projectService;

    @Autowired
    private ServerService serverService;

    @Test
    @Order(1)
    void testCreateProject() {
        ProjectCreateRequest request = new ProjectCreateRequest();
        request.setName("Test Project");
        request.setCode("test-project");
        request.setType(ProjectType.JAVA_MAVEN);

        ProjectDTO project = projectService.create(request);
        assertNotNull(project.getId());
    }

    @Test
    @Order(2)
    void testDeployFlow() {
        // 1. 创建部署请求
        DeployRequest request = new DeployRequest();
        request.setProjectId(1L);
        request.setVersionId(1L);
        request.setEnv("dev");
        request.setStrategy(DeployType.SINGLE);

        // 2. 触发部署
        DeployDTO deploy = deployService.deploy(request);
        assertNotNull(deploy.getId());

        // 3. 等待部署完成 (最多5分钟)
        await().atMost(5, TimeUnit.MINUTES)
            .until(() -> {
                DeployDTO current = deployService.getById(deploy.getId());
                return current.getStatus() == DeployStatus.SUCCESS
                    || current.getStatus() == DeployStatus.FAIL;
            });

        // 4. 验证结果
        DeployDTO result = deployService.getById(deploy.getId());
        assertEquals(DeployStatus.SUCCESS, result.getStatus());
    }

    @Test
    @Order(3)
    void testRollback() {
        DeployDTO lastDeploy = deployService.list(...).getRecords().get(0);

        RollbackRequest request = new RollbackRequest();
        request.setReason("Test rollback");

        deployService.rollback(lastDeploy.getId(), request);

    }

### 3.3 Nginx 配置

# drp-ui/nginx.conf
server {
    listen 80;
    server_name _;

    root /usr/share/nginx/html;
    index index.html;

    # 前端静态资源
    location / {
        try_files $uri $uri/ /index.html;
    }

    # API 代理
    location /api/ {
        proxy_pass http://drp-gateway:8080/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # WebSocket 代理
    location /ws/ {
        proxy_pass http://drp-gateway:8080/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }
}

## 4. 部署脚本

### 4.1 一键部署脚本

```bash
#!/bin/bash
# deploy.sh - DRP Platform 一键部署脚本

set -e

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

# 配置
COMPOSE_FILE="docker-compose.yml"
IMAGE_TAG=${1:-latest}

echo -e "${GREEN}========================================${NC}"
echo -e "${GREEN}   DRP Platform 部署脚本 v1.0.0${NC}"
echo -e "${GREEN}========================================${NC}"

# 检查 Docker
if ! command -v docker &> /dev/null; then
    echo -e "${RED}错误: Docker 未安装${NC}"
    exit 1
fi

# 检查 docker-compose
if ! command -v docker-compose &> /dev/null; then
    echo -e "${RED}错误: docker-compose 未安装${NC}"
    exit 1
fi

# 创建环境变量文件
create_env_file() {
    if [ ! -f .env ]; then
        echo -e "${YELLOW}创建环境变量文件...${NC}"
        cat > .env << EOF
MYSQL_ROOT_PASSWORD=DrpRoot2024!
MYSQL_PASSWORD=DrpDb2024!
REDIS_PASSWORD=DrpRedis2024!
MINIO_ROOT_USER=drpadmin
MINIO_ROOT_PASSWORD=DrpMinio2024!
JWT_SECRET=your-256-bit-secret-key-here-must-be-at-least-32-chars
ENCRYPT_KEY=your-32-character-encrypt-key
EOF
        echo -e "${GREEN}环境变量文件已创建${NC}"
    fi
}

# 构建镜像
build_images() {
    echo -e "${YELLOW}构建后端服务镜像...${NC}"
    cd drp-platform
    mvn clean package -DskipTests
    cd ..

    echo -e "${YELLOW}构建前端...${NC}"
    cd drp-ui
    npm run build
    cd ..
}

# 启动服务
start_services() {
    echo -e "${YELLOW}启动服务...${NC}"
    docker-compose up -d

    echo -e "${YELLOW}等待服务启动...${NC}"
    sleep 30

    # 检查服务状态
    docker-compose ps
}

# 初始化数据库
init_database() {
    echo -e "${YELLOW}初始化数据库...${NC}"
    docker-compose exec -T mysql mysql -uroot -p${MYSQL_ROOT_PASSWORD} drp < init.sql
}

# 主流程
main() {
    create_env_file
    build_images
    start_services
    init_database

    echo -e "${GREEN}========================================${NC}"
    echo -e "${GREEN}   部署完成!${NC}"
    echo -e "${GREEN}   访问地址: http://localhost${NC}"
    echo -e "${GREEN}   默认账号: admin / admin123${NC}"
    echo -e "${GREEN}========================================${NC}"
}

main "$@"
```

## 5. 用户手册

### 5.1 快速开始

```markdown
# DRP 平台快速开始

## 1. 登录系统
访问 http://localhost，使用默认账号登录:
- 用户名: admin
- 密码: admin123

## 2. 添加服务器
1. 进入「服务器管理」菜单
2. 点击「添加服务器」
3. 填写服务器信息:
   - 服务器名称: 生产服务器-01
   - 主机地址: 192.168.1.100
   - SSH端口: 22
   - 用户名: root
   - 认证方式: 密码/私钥
4. 点击「测试连接」验证
5. 点击「确定」保存

## 3. 创建项目
1. 进入「项目管理」菜单
2. 点击「新建项目」
3. 填写项目信息:
   - 项目名称: 用户服务
   - 项目编码: user-service
   - 项目类型: Java Maven
   - Git仓库: https://github.com/xxx/user-service.git
4. 配置分支策略 (如: master, develop)
5. 点击「确定」创建

## 4. 配置环境变量
1. 进入项目详情
2. 点击「环境变量」tab
3. 添加变量:
   - 环境: dev
   - 变量名: DB_URL
   - 变量值: jdbc:mysql://localhost:3306/user
   - 是否敏感: 否

## 5. 触发构建
1. 进入「构建管理」菜单
2. 点击「触发构建」
3. 选择项目
4. 选择分支
5. 点击「确定」

## 6. 部署
1. 构建成功后，点击「部署」
2. 选择目标环境 (dev/test/prod)
3. 选择部署策略 (单机/蓝绿)
4. 点击「开始部署」
5. 查看部署进度和日志
6. 等待部署完成

## 7. 回滚
1. 进入「部署记录」
2. 找到要回滚的版本
3. 点击「回滚」
4. 确认回滚
```

### 5.2 常见问题

```markdown
# 常见问题

## Q1: 部署失败，日志显示 "Connection refused"
A: 检查目标服务器的SSH服务是否启动，端口是否正确，防火墙是否放行。

## Q2: 健康检查超时
A: 1. 检查应用启动时间，适当增加健康检查超时时间
   2. 检查健康检查URL是否正确
   3. 查看应用启动日志排查原因

## Q3: 构建产物上传失败
A: 1. 检查MinIO服务是否正常运行
   2. 检查网络连接
   3. 查看磁盘空间是否充足

## Q4: 回滚失败
A: 1. 检查备份文件是否存在
   2. 检查备份恢复脚本是否正确
   3. 查看回滚日志定位问题
```
