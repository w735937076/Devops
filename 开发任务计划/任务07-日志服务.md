## 5. 操作日志设计

### 5.1 操作日志注解

```java
// com.drp.log.annotation.OperationLog
@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
@Documented
public @interface OperationLog {
    String value();                           // 操作名称
    OperationType type();                    // 操作类型
    String projectParam() default "";       // 项目ID参数名
    String envParam() default "";           // 环境参数名
    String versionParam() default "";       // 版本参数名
}
```

### 5.2 操作日志切面

```java
// com.drp.log.aspect.OperationLogAspect
@Aspect
@Component
@RequiredArgsConstructor
@Slf4j
public class OperationLogAspect {

    private final OperationLogService operationLogService;

    @Around("@annotation(operationLog)")
    public Object around(ProceedingJoinPoint point, com.drp.log.annotation.OperationLog operationLog) throws Throwable {
        HttpServletRequest request = getHttpServletRequest();
        Long operatorId = getCurrentUserId();
        String operator = getCurrentUsername();

        long start = System.currentTimeMillis();
        Object result = null;
        boolean success = true;
        String errorMessage = null;

        try {
            result = point.proceed();
            return result;
        } catch (Exception e) {
            success = false;
            errorMessage = e.getMessage();
            throw e;
        } finally {
            long duration = System.currentTimeMillis() - start;

            // 异步记录日志
            CompletableFuture.runAsync(() -> {
                try {
                    OperationLogCreateRequest createRequest = new OperationLogCreateRequest();
                    createRequest.setOperator(operator);
                    createRequest.setOperatorId(operatorId);
                    createRequest.setOperationType(operationLog.type());
                    createRequest.setProjectId(getParamValue(point, operationLog.projectParam()));
                    createRequest.setEnv(getParamValue(point, operationLog.envParam()));
                    createRequest.setVersion(getParamValue(point, operationLog.versionParam()));
                    createRequest.setDetail(getMethodParams(point));
                    createRequest.setIp(getClientIp(request));
                    createRequest.setStatus(success ? "SUCCESS" : "FAIL");
                    createRequest.setDuration(duration);
                    createRequest.setErrorMessage(errorMessage);

                    operationLogService.create(createRequest);
                } catch (Exception e) {
                    log.error("记录操作日志失败", e);
                }
            });
        }
    }

    private String getClientIp(HttpServletRequest request) {
        String ip = request.getHeader("X-Forwarded-For");
        if (StringUtils.isBlank(ip) || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getHeader("Proxy-Client-IP");
        }
        if (StringUtils.isBlank(ip) || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getHeader("WL-Proxy-Client-IP");
        }
        if (StringUtils.isBlank(ip) || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getRemoteAddr();
        }
        return ip;
    }
}
```

### 5.3 操作日志服务

```java
// com.drp.log.service.OperationLogService
@Service
@RequiredArgsConstructor
public class OperationLogServiceImpl implements OperationLogService {

    private final OperationLogRepository repository;

    @Override
    public PageResponse<OperationLogDTO> list(LogQueryRequest request) {
        // 构建查询条件
        Specification<OperationLog> spec = Specification.where(null);

        if (request.getOperatorId() != null) {
            spec = spec.and((root, query, cb) ->
                cb.equal(root.get("operatorId"), request.getOperatorId()));
        }
        if (request.getProjectId() != null) {
            spec = spec.and((root, query, cb) ->
                cb.equal(root.get("projectId"), request.getProjectId()));
        }
        if (StringUtils.isNotBlank(request.getOperationType())) {
            spec = spec.and((root, query, cb) ->
                cb.equal(root.get("operationType"), request.getOperationType()));
        }
        if (request.getStartTime() != null) {
            spec = spec.and((root, query, cb) ->
                cb.greaterThanOrEqualTo(root.get("createTime"), request.getStartTime()));
        }
        if (request.getEndTime() != null) {
            spec = spec.and((root, query, cb) ->
                cb.lessThanOrEqualTo(root.get("createTime"), request.getEndTime()));
        }

        Page<OperationLog> page = repository.findAll(spec,
            PageRequest.of(request.getPage() - 1, request.getSize(),
                Sort.by(Sort.Direction.DESC, "createTime")));

        List<OperationLogDTO> records = page.getContent().stream()
            .map(this::toDTO)
            .collect(Collectors.toList());

        return PageResponse.of(records, page.getTotalElements(),
            page.getNumber() + 1, page.getSize());
    }

    @Override
    @Transactional
    public void create(OperationLogCreateRequest request) {
        OperationLog log = new OperationLog();
        BeanUtils.copyProperties(request, log);
        repository.save(log);
    }
}
```

## 6. 应用日志实时查看

### 6.1 WebSocket 处理器

```java
// com.drp.log.websocket.AppLogHandler
@Component
@RequiredArgsConstructor
public class AppLogHandler {

    private final AppLogSessionManager sessionManager;
    private final SshService sshService;
    private final ServerRepository serverRepository;

    @Override
    public void handleTextMessage(WebSocketSession session, TextMessage message) {
        Long serverId = extractServerId(session);
        String command = message.getPayload();

        try {
            Server server = serverRepository.findById(serverId).orElseThrow();

            // 执行 tail -f 命令
            executeTailLog(server, command, session);
        } catch (Exception e) {
            sendError(session, e.getMessage());
        }
    }

    private void executeTailLog(Server server, String logPath, WebSocketSession session) {
        String command = String.format("tail -f %s", logPath);

        // 使用异步方式持续读取日志
        Executors.newSingleThreadExecutor().submit(() -> {
            try {
                Process process = Runtime.getRuntime().exec(command);
                BufferedReader reader = new BufferedReader(
                    new InputStreamReader(process.getInputStream()));

                String line;
                while ((line = reader.readLine()) != null) {
                    if (session.isOpen()) {
                        session.sendMessage(new TextMessage(line));
                    } else {
                        break;
                    }
                }
            } catch (Exception e) {
                try {
                    sendError(session, e.getMessage());
                } catch (Exception ignored) {}
            }
        });
    }

    private void sendError(WebSocketSession session, String message) throws IOException {
        session.sendMessage(new TextMessage("[ERROR] " + message));
    }
}
```

```vue
async function loadLogs() {
  loading.value = true
  try {
    const params = {
      ...query,
      startTime: query.dateRange?.[0],
      endTime: query.dateRange?.[1]
    }
    const res = await logApi.listOperations(params)
    logs.value = res.records
    total.value = res.total
  } finally {
    loading.value = false
  }
}

function handleRowClick(row: any) {
  currentLog.value = row
  detailVisible.value = true

function handleExport() {
  window.open('/api/logs/operations/export?' + new URLSearchParams(query).toString())
}

function formatDetail(detail: any): string {
  return JSON.stringify(detail, null, 2)
}

onMounted(() => {
  loadLogs()
})
```
